    # webpack.config.js

    采用commonjs语法格式

    const {resolve} = require('path')
const pluginA =require('plugin-a')
const pluginB =require('plugin-b')
module.exports = {
	entry:'./src/index.js',						//入口文件
	output:{
		filename:'built.js',
		path:resolve(__dirname,'build') //输出文件
	},
	module:{
		rules:[{...},{...},...]
	},
	plugins:[ new pluginA(),new pluginB(),...],
	mode:'development',
//mode:'production'
//生产模式中打包文件代码被压缩,缩小打包文件的体积
}

    # url-loader

    # file-loader

    处理引入的文件

    file-loader默认情况下，生成的文件的文件名就是文件内容的 MD5 哈希值并会保留所引用资源的原始扩展名

    url-loader 功能类似于 file-loader，但是在文件大小（单位 byte）低于指定的限制时，可以返回一个 DataURL。

    module:{
	rules:[
		{
      test:/\.(jpg|png|gif)$/,
      loader:'url-loader',					//需下载url-loader和file-loader
      options:{
        limit:8*1024
      }
    }
	]
}

    # html-loader

    处理html文件中img,搭配url-loader和file-loader使用

    module:{
	rules:[
		{
			test:/\.html$/,
			loader:'html-loader',
		}
	]
}

    # 踩坑

    1. 问题描述：webpack 打包出现 Error: Automatic publicPath is not supported in this browser

      解决方法： 在webpack.config.js文件中添加

      module.exports = 
{
   devserver:{
        contentBase: resolve(__dirname, 'build'),	      //启动路径
        compress: true,																	//是否开启gzip压缩
        port: 8888,																			//设置端口号
     		open:true																				//打包完成后是否自动打开浏览器
   }
}

    2. 问题描述 :终端中执行webpack-dev-server出现

      Error: Cannot find module 'webpack-cli/bin/config-yargs'

      解决方法:

      在终端中直接使用

      webpack server

      or

      webpack s
